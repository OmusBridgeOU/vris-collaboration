    locale: 'ja',
    messages: {
      ja: { list: ['a', 'b', 'c'] },
      en: { list: ['a', 'b', 'c'] },
    },
  })

      setup: () => {
        const i18n = useI18n()

      },
    }),
    {
      global: {
        plugins: [i18n],
      },
    },
  )
})
